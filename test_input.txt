import { Component } from '@angular/core';
import { CommonModule } from '@angular/common';
import { FormsModule } from '@angular/forms';
import { RouterModule } from '@angular/router';
import { DashboardShellComponent } from '../../../components/dashboard/dashboard-shell.component';
import { StudentNavComponent } from '../../../components/dashboard/student-nav.component';

interface TaskItem {
  id: string;
  text: string;
  done: boolean;
}

@Component({
  selector: 'app-student-dashboard-page',
  standalone: true,
  imports: [CommonModule, RouterModule, FormsModule, DashboardShellComponent, StudentNavComponent],
  templateUrl: './student-dashboard.component.html',
})
export class StudentDashboardPageComponent {
  studentName = 'Student';
  tasks: TaskItem[] = [];
  newTask = '';
  private readonly storageKey = 'student-dashboard-tasks';

  constructor() {
    const rawSession = localStorage.getItem('userSession');

    if (!rawSession) {
      return;
    }

    try {
      const session = JSON.parse(rawSession) as { name?: string };

      if (session?.name?.trim()) {
        this.studentName = session.name.trim();
      }
    } catch {
      // Keep default name if session JSON is invalid.
    }

    this.loadTasks();
  }

  addTask(): void {
    const text = this.newTask.trim();
    if (!text) {
      return;
    }

    const id =
      typeof crypto !== 'undefined' && 'randomUUID' in crypto
        ? crypto.randomUUID()
        : `${Date.now()}-${Math.random().toString(16).slice(2)}`;

    this.tasks = [{ id, text, done: false }, ...this.tasks];
    this.newTask = '';
    this.saveTasks();
  }

  get remainingTasks(): number {
    return this.tasks.filter((task) => !task.done).length;
  }

  toggleTask(taskId: string): void {
    this.tasks = this.tasks.map((task) =>
      task.id === taskId ? { ...task, done: !task.done } : task,
    );
    this.saveTasks();
  }

  removeTask(taskId: string): void {
    this.tasks = this.tasks.filter((task) => task.id !== taskId);
    this.saveTasks();
  }

  private loadTasks(): void {
    if (typeof window === 'undefined' || typeof localStorage === 'undefined') {
      return;
    }

    try {
      const raw = localStorage.getItem(this.storageKey);
      if (!raw) {
        return;
      }
      const parsed = JSON.parse(raw);
      if (Array.isArray(parsed)) {
        this.tasks = parsed.filter(
          (item): item is TaskItem =>
            typeof item?.id === 'string' && typeof item?.text === 'string',
        );
      }
    } catch {
      // Ignore invalid localStorage data.
    }
  }

  private saveTasks(): void {
    if (typeof window === 'undefined' || typeof localStorage === 'undefined') {
      return;
    }

    try {
      localStorage.setItem(this.storageKey, JSON.stringify(this.tasks));
    } catch {
      // Ignore storage write errors.
    }
  }
}

import { Component } from '@angular/core';

type Section = 'none' | 'active' | 'reviews' | 'messages' | 'manage';

@Component({
  selector: 'app-counselor-dashboard',
  templateUrl: './counselor-dashboard.component.html',
  imports: [
    DashboardShellComponent,
    CounselorNavComponent,
    RouterModule,
    CommonModule,
    ManageStudentsComponent,
  ],
})
export class CounselorDashboardPageComponent {
  managingStudentsCount = 0;
  managingStudentsLoading = true;

  private sectionSubject = new BehaviorSubject<Section>('manage');
  section$ = this.sectionSubject.asObservable();

  toggleSection(sec: Section) {
    const current = this.sectionSubject.value;
    this.sectionSubject.next(current === sec ? 'none' : sec);
  }

  onStudentsLoaded(count: number) {
    this.managingStudentsCount = count;
    this.managingStudentsLoading = false;
  }

  onStudentsLoading(isLoading: boolean) {
    this.managingStudentsLoading = isLoading;
  }
}

  id: number;
  studentId: number;
  student: UserData | null;

  permissionLevel: number;
  createdAt: string;
  expiresAt: string;
}

interface StudentProfile {
  studentId: number;
  student: UserData | null;
  contentBlockIds: number[];
  permissionLevels: number[];
  shares: StudentShare[];
}

export class ManageStudentsComponent implements OnInit {
  studentShares: StudentProfile[] = [];
  loading = false;
  error: string | null = null;

  @Output() studentsLoaded = new EventEmitter<number>();
  @Output() loadingStateChanged = new EventEmitter<boolean>();

  constructor(
    private shareBlocksService: ShareBlocksService,
    private userService: UserService,
    private router: Router,
    private cdr: ChangeDetectorRef,
  ) {}

  ngOnInit() {
    this.loadStudents();
  }

  loadStudents() {
    this.loading = true;
    this.loadingStateChanged.emit(true);
    this.error = null;
    this.studentShares = [];

    this.shareBlocksService.getSharedStudents().subscribe({
      next: (response) => {
        if (response.succeeded) {
          const shares = response.data;

          // Group shares by studentId so all block IDs stay together for each student
          const sharesByStudentId = new Map<number, ShareResponseDTO[]>();
          shares.forEach((share) => {
            const existingShares = sharesByStudentId.get(share.sharedByUserId) ?? [];
            existingShares.push(share);
            sharesByStudentId.set(share.sharedByUserId, existingShares);
          });

          const uniqueStudentIds = Array.from(sharesByStudentId.keys());

          // Fetch user details for each unique student
          let completedFetches = 0;
          const totalFetches = uniqueStudentIds.length;

          if (totalFetches === 0) {
            this.loading = false;
            this.loadingStateChanged.emit(false);
            this.studentsLoaded.emit(0);
            this.cdr.detectChanges();
            return;
          }

          const studentShareMap = new Map<number, StudentProfile>();

          uniqueStudentIds.forEach((studentId) => {
            this.userService.getUserById(studentId).subscribe({
              next: (userResponse) => {
                if (userResponse.succeeded && userResponse.data.length > 0) {
                  const student = userResponse.data[0];
                  const studentSharesForStudent = sharesByStudentId.get(studentId) ?? [];

                  studentShareMap.set(studentId, {
                    studentId,
                    student,
                    contentBlockIds: studentSharesForStudent.map((share) => share.contentBlockId),
                    permissionLevels: studentSharesForStudent.map((share) => share.permissionLevel),
                    shares: studentSharesForStudent.map((share) => ({
                      id: share.id,
                      studentId,
                      student,
                      contentBlockId: share.contentBlockId,
                      permissionLevel: share.permissionLevel,
                      createdAt: share.createdAt,
                      expiresAt: share.expiresAt,
                    })),
                  });
                }

                completedFetches++;
                if (completedFetches === totalFetches) {
                  this.studentShares = Array.from(studentShareMap.values());
                  this.loading = false;
                  this.loadingStateChanged.emit(false);
                  this.studentsLoaded.emit(this.studentShares.length);
                  this.cdr.detectChanges();
                }
              },
              error: (err) => {
                console.error(`Error fetching user details for student ${studentId}:`, err);
                completedFetches++;
                if (completedFetches === totalFetches) {
                  this.studentShares = Array.from(studentShareMap.values());
                  this.loading = false;
                  this.loadingStateChanged.emit(false);
                  this.studentsLoaded.emit(this.studentShares.length);
                  this.cdr.detectChanges();
                }
              },
            });
          });
        } else {
          this.loading = false;
          this.loadingStateChanged.emit(false);
          this.studentsLoaded.emit(0);
          this.cdr.detectChanges();
        }
      },
      error: (err) => {
        console.error('Error fetching shared students:', err);
        this.error = 'Failed to load shared students. Please try again.';
        this.loading = false;
        this.loadingStateChanged.emit(false);
        this.studentsLoaded.emit(0);
        this.cdr.detectChanges();
      },
    });
  }

  viewStudentProfile(studentProfile: StudentProfile) {
    if (!studentProfile.student) {
      alert('Student information not available');
      return;
    }

    const sharesForStudent = studentProfile.shares;

    // Keep every shared content block ID for this student
    const contentBlockIds = studentProfile.contentBlockIds.join(',');

    const permissionMap = sharesForStudent.reduce(
      (acc, share) => {
        acc[share.contentBlockId] = share.permissionLevel;
        return acc;
      },
      {} as Record<number, number>,
    );

    const studentName = `${studentProfile.student.firstName} ${studentProfile.student.lastName}`;

    console.log('🔍 Navigating to student profile:', {
      studentId: studentProfile.studentId,
      studentName,
      contentBlockIds,
      permissionMap,
    });

    this.router.navigate(['/counselor/view-student'], {
      queryParams: {
        studentId: studentProfile.studentId,
        studentName,
        contentBlockIds,
        permissionMap: JSON.stringify(permissionMap),
      },
    });
  }

  getPermissionSummary(studentProfile: StudentProfile): string {
    const labels = Array.from(
      new Set(studentProfile.permissionLevels.map((level) => this.getPermissionLabel(level))),
    );

    return labels.length > 0 ? labels.join(' / ') : 'No permission';
  }

  getPermissionBadgeClass(studentProfile: StudentProfile): string {
    const uniquePermissionLevels = Array.from(new Set(studentProfile.permissionLevels));

    if (uniquePermissionLevels.length === 1) {
      switch (uniquePermissionLevels[0]) {
        case 1:
          return 'bg-blue-100 text-blue-800';
        case 2:
          return 'bg-yellow-100 text-yellow-800';
        case 3:
          return 'bg-green-100 text-green-800';
        default:
          return 'bg-gray-100 text-gray-800';
      }
    }

    return 'bg-slate-100 text-slate-800';
  }

  private getPermissionLabel(permissionLevel: number): string {
    switch (permissionLevel) {
      case 1:
        return 'View';
      case 2:
        return 'Comment';
      case 3:
        return 'Edit';
      default:
        return 'Unknown';
    }
  }
}
